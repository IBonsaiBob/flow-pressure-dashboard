' VBA_ThisWorkbook.txt  -  ThisWorkbook module
' v1.73
'
' Paste this entire file into the ThisWorkbook module in the
' VBA editor (Alt+F11, double-click "ThisWorkbook" in the
' project tree).
'
' PURPOSE
'   Application.OnKey is APPLICATION-level, not workbook-level.
'   Registering a shortcut in Workbook_Open means it stays active
'   across every other open workbook until Excel is closed.
'
'   Instead we register on Workbook_Activate (this workbook
'   becomes the active one) and unregister on Workbook_Deactivate
'   (user switches away) and Workbook_BeforeClose (file closes).
'   This scopes the custom shortcut to exactly when this
'   workbook is in focus.
'
' SHORTCUT
'   Ctrl+Shift+V  →  PasteValuesDispatch (defined below)
'     • On the Plots sheet: copies selected sensor codes into B6:K6
'       (CopySelectionToPlotsHeaders).
'     • On any other sheet: pastes clipboard as values only.
'
' NOTE
'   Workbook_Open / Workbook_Activate also reset Ctrl+V to the
'   Excel default to clear any stale OnKey binding that may have
'   been left by an older version of this file.
' ============================================================
Option Explicit

Private Sub Workbook_Open()
    ' Reset any stale Ctrl+V binding from older versions, then register
    ' Ctrl+Shift+V for this workbook.
    Application.OnKey "^V"
    Application.OnKey "^+V", "PasteValuesDispatch"
End Sub

Private Sub Workbook_Activate()
    ' Re-register whenever the user switches back to this workbook.
    Application.OnKey "^V"
    Application.OnKey "^+V", "PasteValuesDispatch"
End Sub

Private Sub Workbook_Deactivate()
    ' Unregister when user switches to another workbook —
    ' restores Ctrl+Shift+V to Excel default in all other files.
    Application.OnKey "^+V"
End Sub

Private Sub Workbook_BeforeClose(Cancel As Boolean)
    ' Unregister before the file closes so the shortcut doesn't
    ' linger if the user cancels the close and works elsewhere.
    Application.OnKey "^+V"
End Sub

' ---------------------------------------------------------------------------
' PasteValuesDispatch
' Bound to Ctrl+Shift+V (registered above).
' Context-sensitive dispatcher:
'   • On the Plots sheet  → calls CopySelectionToPlotsHeaders (copies selected
'     sensor codes into B6:K6 and refreshes the chart).
'   • On any other sheet  → pastes the clipboard as values only, equivalent to
'     Home > Paste > Paste Values (xlPasteValues).
'
' WHY THE FALLBACK IS NEEDED
'   Excel automatically clears "copy mode" (marching ants) whenever any VBA
'   code writes to a cell.  Switching to the Total Head tab triggers
'   Worksheet_Activate → RebuildDataTable, which writes cells and kills copy
'   mode.  After that, PasteSpecial raises error 1004 silently.
'   The fallback reads tab-delimited text straight from the Windows OS
'   clipboard (the same source Win+V uses) so the paste always works,
'   regardless of which sheet the data was copied from.
' ---------------------------------------------------------------------------
Public Sub PasteValuesDispatch()
    If ActiveSheet.Name = "Plots" Then
        CopySelectionToPlotsHeaders
        Exit Sub
    End If

    ' ── Try native paste-values first ──────────────────────────────────────
    ' Works when cells are still in copy-mode (marching ants present), e.g.
    ' when the user copies and pastes on the same sheet without activating
    ' another sheet in between.
    On Error Resume Next
    Selection.PasteSpecial Paste:=xlPasteValues
    Dim pasteErr As Long: pasteErr = Err.Number
    On Error GoTo 0
    If pasteErr = 0 Then Exit Sub

    ' ── Fallback: read raw text from the Windows OS clipboard ──────────────
    ' PasteSpecial raised error 1004 — copy mode was cleared (most likely by
    ' Worksheet_Activate writing cells).  Read the clipboard text directly,
    ' parse it as tab-delimited rows, and write values into the selection,
    ' coercing numeric strings to numbers where possible.
    Dim clipText As String
    On Error Resume Next
    clipText = CreateObject("htmlfile").ParentWindow.ClipboardData.GetData("text")
    On Error GoTo 0
    If Len(Trim(clipText)) = 0 Then Exit Sub

    ' Normalise line endings and strip any trailing newline.
    clipText = Replace(clipText, vbCrLf, vbLf)
    clipText = Replace(clipText, vbCr, vbLf)
    If Right(clipText, 1) = vbLf Then _
        clipText = Left(clipText, Len(clipText) - 1)

    Dim rowArr() As String: rowArr = Split(clipText, vbLf)
    Dim startCell As Range: Set startCell = Selection.Cells(1, 1)
    Dim i As Long, j As Long, colArr() As String, cv As String
    For i = 0 To UBound(rowArr)
        colArr = Split(rowArr(i), vbTab)
        For j = 0 To UBound(colArr)
            cv = colArr(j)
            If IsNumeric(cv) And Len(Trim(cv)) > 0 Then
                startCell.Offset(i, j).Value = CDbl(cv)
            Else
                startCell.Offset(i, j).Value = cv
            End If
        Next j
    Next i
End Sub

' ---------------------------------------------------------------------------
' CopySelectionToPlotsHeaders
' Called by PasteValuesDispatch when on the Plots sheet.
' Takes the currently selected cells (from any sheet), extracts up to 10
' non-empty text values, writes them sequentially into B6:K6 on the Plots
' sheet, then triggers a data lookup for each column and refreshes the chart.
' This lets the user select sensor codes in Point Index (or anywhere else)
' and paste them straight into the Plots header row in one keystroke.
' ---------------------------------------------------------------------------
Private Sub CopySelectionToPlotsHeaders()
    Dim ws As Worksheet
    On Error Resume Next
    Set ws = ThisWorkbook.Sheets("Plots")
    On Error GoTo 0
    If ws Is Nothing Then Exit Sub

    ' Capture the current selection before potentially activating another sheet.
    Dim sel As Range
    On Error Resume Next
    Set sel = Application.Selection
    On Error GoTo 0
    If sel Is Nothing Then Exit Sub

    ' Collect up to 10 non-empty string values from the selected cells.
    Dim vals(0 To 9) As String
    Dim count As Integer: count = 0
    Dim cel As Range
    For Each cel In sel.Cells
        If count >= 10 Then Exit For
        Dim v As String: v = Trim(CStr(cel.Value))
        If v <> "" Then
            vals(count) = v
            count = count + 1
        End If
    Next cel
    If count = 0 Then Exit Sub

    ' Write codes into B6:K6 and trigger lookup + chart refresh.
    Application.EnableEvents = False
    Dim col As Long
    For col = 0 To count - 1
        ws.Cells(6, 2 + col).Value = vals(col)
    Next col
    Application.EnableEvents = True

    For col = 0 To count - 1
        Call LookupAndFillSensorColumn(2 + col)
    Next col
    Call RefreshPlotsChart

    ' Switch to the Plots sheet so the user sees the result immediately.
    ws.Activate
End Sub
