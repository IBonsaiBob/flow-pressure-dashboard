' ============================================================
' VBA_Module3_TotalHead.txt  -  Module3  (Total Head logic)
' v1.79
'
' KEY FIXES vs. previous broken version:
'   1. ArchiveTotalHeadData – rewrote to use the correct archive
'      column layout and a reliable zone-name search so the save
'      no longer clobbers unrelated rows or silently fails.
'   2. LoadTotalHeadByName – optimised so each source tab is read
'      exactly once (same cost as typing a single Point Ref, just
'      spread across 3 tabs: Archive, Pressure Data, Point Index).
'      Before the fix every sub called FindPressureDataCol / PI
'      lookups individually, so 35 sensors meant 35 full scans
'      of each sheet.
'   3. (v1.73) FindPressureDataCol – changed to case-insensitive
'      comparison and added Logger ID fallback.  Previously an
'      exact case mismatch (e.g. PD header "prv06-ds" vs config
'      "PRV06-DS") or a PD sheet indexed by Logger ID instead of
'      Point Ref caused every sensor to return pdCol=0 silently,
'      producing headers in the data table but empty data columns.
'      Also added a status-bar message after RebuildDataTable
'      reporting how many sensors were matched vs missed.
'   4. (v1.74) RebuildDataTable – when ALL configured sensors fail
'      to match (sensorsFound=0, sensorsTotal>0) a MsgBox now shows
'      which Point Refs were expected and what headers exist in PD
'      row 7.  This replaces the silent status-bar-only feedback
'      that was easy to miss, making the mismatch immediately
'      obvious so the user can fix the PD headers or Point Refs.
'   5. (v1.75) Added Public Const MODULE3_VERSION = "1.75" so that
'      Workbook_Open (ThisWorkbook v1.75) can verify on startup that
'      the in-workbook Module3 matches the expected repository
'      version, alerting the user if stale code is detected.
'   6. (v1.76) Fixed PI_COL_LOGGER_ID: was 18 (col R) but Logger ID
'      is actually at col O = 15 in the Point Index.  The v1.73
'      Logger ID fallback therefore always read empty cells and never
'      matched anything — corrected to 15.
'   7. (v1.77) Added GPS / Dip / Dumpy bidirectional sync between
'      TH cols C/E/F and Point Index cols L/J/K (same logic as the
'      existing Final Z / col D ↔ col M sync).  Fixed Notes column
'      reference (col G→H, i.e. 7→8) after a new GPS column was
'      inserted in TH.
'   8. (v1.78) Fixed three bugs:
'      a) PopulateOneSensor now uses addition (pressure + offset +
'         Final Z) not multiplication.  Default finalZ changed from
'         1 → 0 so sensors with no elevation give pressure+offset.
'      b) Clear-ref block now clears GPS/Dip/Dumpy/Z/Offset BEFORE
'         calling RecordSensorNote so they are always wiped even if
'         note-recording fails.
'      c) mOrigRefByRow / mOrigZByRow are snapshotted in
'         Worksheet_Activate so clearing a ref after a zone-load
'         (before the user has clicked the cell) still saves the
'         note to Point Index.
'   9. (v1.79) Fixed cells B9:F43 not clearing after a zone load:
'      a) LoadTotalHeadByName clear loop set Offset to 0 instead of
'         "" so rows without a Point Ref showed "0.000" in col B.
'         Changed to "".
'      b) Added a safety pass after the restore loop that explicitly
'         clears cols B-F (Offset/GPS/Z/Dip/Dumpy/Notes) for every
'         config row whose Point Ref is still empty after loading,
'         eliminating any stale data left from a previous session.
'      c) Worksheet_Change clear-ref block: Offset was also set to 0
'         instead of "" — fixed for consistency.
'
' SHEET LAYOUT (Total Head)

'   D5  row 5, col 4   Start Date  (merged D5:E5)
'   D6  row 6, col 4   End Date    (merged D6:E6)

' ============================================================
' EnsureArchiveHeaders
' ============================================================
Private Sub EnsureArchiveHeaders(wsArch As Worksheet)
    If wsArch.Cells(1, ARCH_COL_ARCHIVED).Value = "" Then _
        wsArch.Cells(1, ARCH_COL_ARCHIVED).Value = "Archived"
    If wsArch.Cells(1, ARCH_COL_SYSTEM).Value = "" Then _
        wsArch.Cells(1, ARCH_COL_SYSTEM).Value = "System"
    If wsArch.Cells(1, ARCH_COL_ZONE).Value = "" Then _
        wsArch.Cells(1, ARCH_COL_ZONE).Value = "Chart Title"
    If wsArch.Cells(1, ARCH_COL_DRV_HEAD).Value = "" Then _
        wsArch.Cells(1, ARCH_COL_DRV_HEAD).Value = "Driving Head"
    If wsArch.Cells(1, ARCH_COL_NOTES).Value = "" Then _
        wsArch.Cells(1, ARCH_COL_NOTES).Value = "Zone Notes"

    Dim s As Long
    For s = 1 To MAX_SENSORS
        Dim ac As Long
        ac = ARCH_COL_SENSOR1 + s - 1
        If wsArch.Cells(1, ac).Value = "" Then
            wsArch.Cells(1, ac).Value = "Sensor " & s
        End If
    Next s
End Sub

' ============================================================
' InitTotalHeadSheet
' ============================================================
Sub InitTotalHeadSheet()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Total Head")
    If ws.Cells(DATA_HDR_ROW, 1).Value = "" Then
        ws.Cells(DATA_HDR_ROW, 1).Value = "Timestamp"
    End If
End Sub

' ============================================================
' ArchiveTotalHeadData  (SAVE)
'
' FIX: previous optimisation attempt broke the zone-search by
' reading the wrong column and miscounting sensor offsets.
' This version:
'   - searches archive col C (ARCH_COL_ZONE) for the current
'     Zone Name; overwrites if found, appends if not
'   - maps config rows 9..43 to archive sensor cols 6..40
'     using the formula  archCol = cfgRow - CFG_FIRST + ARCH_COL_SENSOR1
' ============================================================
Sub ArchiveTotalHeadData()
    Dim wsThis As Worksheet
    Dim wsArch As Worksheet
    On Error GoTo ErrOut

    Set wsThis = ThisWorkbook.Sheets("Total Head")
    Set wsArch = ThisWorkbook.Sheets("Total Head Archive")

    Dim zoneName As String
    zoneName = Trim(CStr(wsThis.Cells(ZONE_ROW, ZONE_COL).Value))
    If zoneName = "" Then
        MsgBox "Enter a Zone Name in cell D3 before saving.", vbExclamation
        Exit Sub
    End If

    Call EnsureArchiveHeaders(wsArch)

    ' ── Find existing row for this zone (search col C) ──────
    Dim archRow As Long
    archRow = 0
    Dim lastRow As Long
    lastRow = wsArch.Cells(wsArch.Rows.Count, ARCH_COL_ZONE).End(xlUp).Row

    Dim r As Long
    For r = 2 To lastRow
        If Trim(CStr(wsArch.Cells(r, ARCH_COL_ZONE).Value)) = zoneName Then
            archRow = r
            Exit For
        End If
    Next r

    If archRow = 0 Then
        archRow = lastRow + 1
        If archRow < 2 Then archRow = 2
    End If

    ' ── Write metadata ───────────────────────────────────────
    wsArch.Cells(archRow, ARCH_COL_ARCHIVED).Value = Now()
    wsArch.Cells(archRow, ARCH_COL_SYSTEM).Value   = wsThis.Cells(SYSTEM_ROW, SYSTEM_COL).Value
    wsArch.Cells(archRow, ARCH_COL_ZONE).Value     = zoneName
    wsArch.Cells(archRow, ARCH_COL_DRV_HEAD).Value = wsThis.Cells(ZONE_ROW, DRV_HEAD_COL).Value
    wsArch.Cells(archRow, ARCH_COL_NOTES).Value    = wsThis.Cells(NOTES_ROW, NOTES_COL).Value

    ' ── Write sensor Point Refs ──────────────────────────────
    ' cfg row 9 → sensor 1 → archive col 6
    ' cfg row 10 → sensor 2 → archive col 7
    ' ...
    ' cfg row 43 → sensor 35 → archive col 40
    Dim cfg As Long
    For cfg = CFG_FIRST To CFG_LAST
        Dim archCol As Long
        archCol = cfg - CFG_FIRST + ARCH_COL_SENSOR1
        wsArch.Cells(archRow, archCol).Value = wsThis.Cells(cfg, 1).Value
    Next cfg

    Application.StatusBar = "Total Head saved: " & zoneName
    Call RefreshLoadDropdown
    Exit Sub

ErrOut:
    MsgBox "ArchiveTotalHeadData error: " & Err.Description, vbCritical
    Application.StatusBar = False
End Sub

' ============================================================
' RefreshLoadDropdown
' Populates validation list on L5 from archive col C.
' Non-critical: if the archive sheet is missing or the cross-sheet
' validation formula is rejected by this Excel version, the sub
' exits silently so that Worksheet_Activate is not interrupted.
' ============================================================
Sub RefreshLoadDropdown()
    Dim wsThis As Worksheet
    Set wsThis = ThisWorkbook.Sheets("Total Head")

    ' Guard: archive sheet may not exist yet on a fresh workbook.
    Dim wsArch As Worksheet
    On Error Resume Next
    Set wsArch = ThisWorkbook.Sheets("Total Head Archive")
    On Error GoTo 0
    If wsArch Is Nothing Then Exit Sub

    Dim lastRow As Long
    lastRow = wsArch.Cells(wsArch.Rows.Count, ARCH_COL_ZONE).End(xlUp).Row

    Dim dropCell As Range
    Set dropCell = wsThis.Cells(SAVE_ROW, LOAD_COL)   ' L5

    With dropCell.Validation
        .Delete
        If lastRow >= 2 Then
            ' Cross-sheet Formula1 references are unsupported in some
            ' Excel versions; suppress any resulting error gracefully.
            On Error Resume Next
            .Add Type:=xlValidateList, _
                 Formula1:="='Total Head Archive'!$C$2:$C$" & lastRow
            .ShowError = False
            On Error GoTo 0
        End If
    End With
End Sub

' ============================================================
' FindPressureDataCol
' Returns the column index in Pressure Data whose row-7 header
' matches ptRef (case-insensitive).  If no exact match is found
' and the mBatch PI cache is loaded, tries the Logger ID from
' Point Index col R as a fallback so that PD sheets indexed by
' Logger ID rather than Point Ref name still work.
' Returns 0 if not found.
' ============================================================
Function FindPressureDataCol(ptRef As String) As Long
    Dim ptRefLC As String
    ptRefLC = LCase(Trim(ptRef))

    If mBatch_Active Then
        Dim j As Long
        ' Pass 1: case-insensitive match on Point Ref name
        For j = 2 To mBatch_pdLastCol
            If LCase(mBatch_pdHdr(j)) = ptRefLC Then
                FindPressureDataCol = j
                Exit Function
            End If
        Next j

        ' Pass 2: Logger ID fallback — look up this sensor's Logger ID
        ' from the PI cache and try matching that against PD headers.
        If mBatch_piCount > 0 Then
            Dim pi As Long
            For pi = 1 To mBatch_piCount
                If LCase(mBatch_piRef(pi)) = ptRefLC Then
                    Dim lgID As String
                    lgID = LCase(mBatch_piLogger(pi))
                    If lgID <> "" Then
                        For j = 2 To mBatch_pdLastCol
                            If LCase(mBatch_pdHdr(j)) = lgID Then
                                FindPressureDataCol = j
                                Exit Function
                            End If
                        Next j
                    End If
                    Exit For   ' found the PI entry; logger scan above exhausted
                End If
            Next pi
        End If

        FindPressureDataCol = 0
        Exit Function
    End If

    ' Non-batch path (called outside RebuildDataTable): case-insensitive scan.
    Dim wsPD As Worksheet
    Set wsPD = ThisWorkbook.Sheets("Pressure Data")
    Dim lastCol As Long
    lastCol = wsPD.Cells(PD_HEADER_ROW, wsPD.Columns.Count).End(xlToLeft).Column
    Dim k As Long
    For k = 2 To lastCol
        If LCase(CStr(wsPD.Cells(PD_HEADER_ROW, k).Value)) = ptRefLC Then
            FindPressureDataCol = k
            Exit Function
        End If
    Next k
    FindPressureDataCol = 0
End Function

' ============================================================
' SyncOffsetToPressureData
' Writes TH col-B offset back to Pressure Data row 5.
' ============================================================
Sub SyncOffsetToPressureData(ptRef As String, offsetVal As Double)
    Dim pdCol As Long
    pdCol = FindPressureDataCol(ptRef)
    If pdCol > 0 Then
        ThisWorkbook.Sheets("Pressure Data").Cells(PD_OFFSET_ROW, pdCol).Value = offsetVal
    End If
End Sub

' ============================================================
' SyncOffsetToTotalHead
' Reads offset from Pressure Data row 5 and writes to TH col B.
' ============================================================
Sub SyncOffsetToTotalHead(ptRef As String, cfgRow As Long)
    Dim pdCol As Long
    pdCol = FindPressureDataCol(ptRef)
    Dim wsThis As Worksheet
    Set wsThis = ThisWorkbook.Sheets("Total Head")

    If pdCol > 0 Then
        Dim ofs As Variant
        If mBatch_Active Then
            ofs = mBatch_pdOfs(pdCol)
        Else
            ofs = ThisWorkbook.Sheets("Pressure Data").Cells(PD_OFFSET_ROW, pdCol).Value
        End If
        wsThis.Cells(cfgRow, 2).Value = IIf(IsNumeric(ofs), CDbl(ofs), 0)
    Else
        wsThis.Cells(cfgRow, 2).Value = 0
    End If
End Sub

' ============================================================
' RepopulateOffsetsFromPressureData
' Reads PD header row ONCE, then sets TH col B for all sensors.
' (Old version called FindPressureDataCol 35 times separately.)
' ============================================================
Sub RepopulateOffsetsFromPressureData()
    Dim wsThis As Worksheet
    Dim wsPD As Worksheet
    Set wsThis = ThisWorkbook.Sheets("Total Head")
    Set wsPD   = ThisWorkbook.Sheets("Pressure Data")

    ' Pre-load PD header + offset rows in a single pass
    Dim lastPDCol As Long
    lastPDCol = wsPD.Cells(PD_HEADER_ROW, wsPD.Columns.Count).End(xlToLeft).Column

    Dim pdHdr() As String
    Dim pdOfs() As Double
    ReDim pdHdr(1 To lastPDCol)
    ReDim pdOfs(1 To lastPDCol)

    Dim c As Long
    For c = 1 To lastPDCol
        pdHdr(c) = CStr(wsPD.Cells(PD_HEADER_ROW, c).Value)
        If IsNumeric(wsPD.Cells(PD_OFFSET_ROW, c).Value) Then
            pdOfs(c) = CDbl(wsPD.Cells(PD_OFFSET_ROW, c).Value)
        End If
    Next c

    ' Apply to each config row using array lookups
    Dim cfg As Long
    For cfg = CFG_FIRST To CFG_LAST
        Dim ptRef As String
        ptRef = Trim(CStr(wsThis.Cells(cfg, 1).Value))
        If ptRef = "" Then GoTo NextCfgOfs

        For c = 2 To lastPDCol
            If LCase(pdHdr(c)) = LCase(ptRef) Then
                wsThis.Cells(cfg, 2).Value = pdOfs(c)
                Exit For
            End If
        Next c
NextCfgOfs:
    Next cfg
End Sub

' ============================================================
' PopulateElevationTH
' Sets TH col D (Z) from Point Index for one config row.
' Does NOT overwrite an existing non-zero numeric Z.
' Accepts optional pre-loaded PI arrays for batch use.
' Pass piData as the array returned by BuildPILookup (Variant()).
' ============================================================
Sub PopulateElevationTH(wsThis As Worksheet, wsPI As Worksheet, _
                         cfgRow As Long, _
                         Optional piRefs As Variant = Empty, _
                         Optional piZ As Variant = Empty, _
                         Optional piCount As Long = 0)
    Dim ptRef As String
    ptRef = Trim(CStr(wsThis.Cells(cfgRow, 1).Value))
    If ptRef = "" Then Exit Sub

    Dim existZ As Variant
    existZ = wsThis.Cells(cfgRow, 4).Value
    If Not IsEmpty(existZ) And IsNumeric(existZ) Then
        If CDbl(existZ) <> 0 Then Exit Sub
    End If

    If piCount > 0 Then
        Dim i As Long
        For i = 1 To piCount
            If piRefs(i) = ptRef Then
                Dim zv As Variant
                zv = piZ(i)
                If Not IsEmpty(zv) And IsNumeric(zv) Then
                    wsThis.Cells(cfgRow, 4).Value = CDbl(zv)
                End If
                Exit Sub
            End If
        Next i
    Else
        Dim lr As Long
        lr = wsPI.Cells(wsPI.Rows.Count, PI_COL_PTREF).End(xlUp).Row
        Dim r As Long
        For r = PI_DATA_FIRST_ROW To lr
            If Trim(CStr(wsPI.Cells(r, PI_COL_PTREF).Value)) = ptRef Then
                Dim zv2 As Variant
                zv2 = wsPI.Cells(r, PI_COL_FINAL_Z).Value
                If Not IsEmpty(zv2) And IsNumeric(zv2) Then
                    wsThis.Cells(cfgRow, 4).Value = CDbl(zv2)
                End If
                Exit Sub
            End If
        Next r
    End If
End Sub

' ============================================================
' PopulateGpsDipDumpyOneTH
' Sets TH cols C (GPS), E (Dip), F (Dumpy) from Point Index
' for one config row.  Does NOT overwrite existing non-zero
' numeric values (same guard as PopulateElevationTH for Z).
' Accepts optional pre-loaded PI arrays for batch use.
' ============================================================
Sub PopulateGpsDipDumpyOneTH(wsThis As Worksheet, cfgRow As Long, _
                              Optional piRefs  As Variant = Empty, _
                              Optional piGps   As Variant = Empty, _
                              Optional piDip   As Variant = Empty, _
                              Optional piDumpy As Variant = Empty, _
                              Optional piCount As Long = 0)
    Dim ptRef As String
    ptRef = Trim(CStr(wsThis.Cells(cfgRow, 1).Value))
    If ptRef = "" Then Exit Sub

    If piCount > 0 Then
        Dim i As Long
        For i = 1 To piCount
            If piRefs(i) = ptRef Then
                Dim gv As Variant : gv = piGps(i)
                Dim dv As Variant : dv = piDip(i)
                Dim dumpyV As Variant : dumpyV = piDumpy(i)
                ' GPS
                Dim exG As Variant : exG = wsThis.Cells(cfgRow, TH_COL_GPS).Value
                If IsEmpty(exG) Or Not IsNumeric(exG) Or CDbl(exG) = 0 Then
                    If Not IsEmpty(gv) And IsNumeric(gv) Then wsThis.Cells(cfgRow, TH_COL_GPS).Value = CDbl(gv)
                End If
                ' Dip
                Dim exD As Variant : exD = wsThis.Cells(cfgRow, TH_COL_DIP).Value
                If IsEmpty(exD) Or Not IsNumeric(exD) Or CDbl(exD) = 0 Then
                    If Not IsEmpty(dv) And IsNumeric(dv) Then wsThis.Cells(cfgRow, TH_COL_DIP).Value = CDbl(dv)
                End If
                ' Dumpy
                Dim exDump As Variant : exDump = wsThis.Cells(cfgRow, TH_COL_DUMPY).Value
                If IsEmpty(exDump) Or Not IsNumeric(exDump) Or CDbl(exDump) = 0 Then
                    If Not IsEmpty(dumpyV) And IsNumeric(dumpyV) Then wsThis.Cells(cfgRow, TH_COL_DUMPY).Value = CDbl(dumpyV)
                End If
                Exit Sub
            End If
        Next i
    Else
        Dim wsPI As Worksheet
        Set wsPI = ThisWorkbook.Sheets("Point Index")
        Dim lr As Long
        lr = wsPI.Cells(wsPI.Rows.Count, PI_COL_PTREF).End(xlUp).Row
        Dim r As Long
        For r = PI_DATA_FIRST_ROW To lr
            If Trim(CStr(wsPI.Cells(r, PI_COL_PTREF).Value)) = ptRef Then
                Dim pgv As Variant : pgv = wsPI.Cells(r, PI_COL_GPS).Value
                Dim pdv As Variant : pdv = wsPI.Cells(r, PI_COL_DIP).Value
                Dim pdumpyV As Variant : pdumpyV = wsPI.Cells(r, PI_COL_DUMPY).Value
                Dim exG2 As Variant : exG2 = wsThis.Cells(cfgRow, TH_COL_GPS).Value
                If IsEmpty(exG2) Or Not IsNumeric(exG2) Or CDbl(exG2) = 0 Then
                    If Not IsEmpty(pgv) And IsNumeric(pgv) Then wsThis.Cells(cfgRow, TH_COL_GPS).Value = CDbl(pgv)
                End If
                Dim exD2 As Variant : exD2 = wsThis.Cells(cfgRow, TH_COL_DIP).Value
                If IsEmpty(exD2) Or Not IsNumeric(exD2) Or CDbl(exD2) = 0 Then
                    If Not IsEmpty(pdv) And IsNumeric(pdv) Then wsThis.Cells(cfgRow, TH_COL_DIP).Value = CDbl(pdv)
                End If
                Dim exDump2 As Variant : exDump2 = wsThis.Cells(cfgRow, TH_COL_DUMPY).Value
                If IsEmpty(exDump2) Or Not IsNumeric(exDump2) Or CDbl(exDump2) = 0 Then

End Sub

' ============================================================
' RecordSensorNote
' Appends a timestamped note to Point Index col V.
' Signature matches v1.48 expectations.
' ============================================================
Sub RecordSensorNote(wsThis As Worksheet, cfgRow As Long, _
                     Optional forceRef As String = "", _
                     Optional origZ As Variant = Empty)
    Dim ptRef As String
    ptRef = IIf(forceRef <> "", Trim(forceRef), _
                Trim(CStr(wsThis.Cells(cfgRow, 1).Value)))
    If ptRef = "" Then Exit Sub

    Dim zVal  As Variant
    Dim ofsVal As Variant
    zVal   = wsThis.Cells(cfgRow, 4).Value
    ofsVal = wsThis.Cells(cfgRow, 2).Value

    Dim zStr  As String
    Dim ofsStr As String
    zStr   = IIf(Not IsEmpty(zVal)   And IsNumeric(zVal),   CStr(CDbl(zVal)), "")
    ofsStr = IIf(Not IsEmpty(ofsVal) And IsNumeric(ofsVal), CStr(CDbl(ofsVal)), "0")

    Dim noteCell As String
    noteCell = CStr(wsThis.Cells(cfgRow, TH_COL_NOTES).Value)

    Dim noteStr As String
    noteStr = Format(Now(), DATE_FMT) & " | Z: " & zStr & _
              ". | Offset: " & ofsStr & ". | Notes: " & noteCell

    ' Include old Z if it has changed
    If Not IsEmpty(origZ) And IsNumeric(origZ) And Not IsEmpty(zVal) And IsNumeric(zVal) Then
        If Abs(CDbl(origZ) - CDbl(zVal)) > Z_TOLERANCE Then
            noteStr = noteStr & ZPI_SUFFIX & " " & CStr(CDbl(origZ))
        End If
    End If

    Dim wsPI As Worksheet
    Set wsPI = ThisWorkbook.Sheets("Point Index")
    Dim lr As Long
    lr = wsPI.Cells(wsPI.Rows.Count, PI_COL_PTREF).End(xlUp).Row

    Dim r As Long
    For r = PI_DATA_FIRST_ROW To lr
        If Trim(CStr(wsPI.Cells(r, PI_COL_PTREF).Value)) = ptRef Then
            Dim existing As String
            existing = CStr(wsPI.Cells(r, PI_COL_NOTES).Value)
            If existing <> "" Then existing = existing & Chr(10)
            wsPI.Cells(r, PI_COL_NOTES).Value = existing & noteStr
            Exit Sub
        End If
    Next r
End Sub

' ============================================================
' RestorePointNotesTH
' Reads the most recent note line from PI col V and puts the
' "Notes: …" fragment back into TH col G.
' Safely skips "Z(PI): …" suffixes.
' ============================================================
Sub RestorePointNotesTH()
    Dim wsThis As Worksheet
    Dim wsPI   As Worksheet
    Set wsThis = ThisWorkbook.Sheets("Total Head")
    Set wsPI   = ThisWorkbook.Sheets("Point Index")

    Dim piLR As Long
    piLR = wsPI.Cells(wsPI.Rows.Count, PI_COL_PTREF).End(xlUp).Row

    Dim cfg As Long
    For cfg = CFG_FIRST To CFG_LAST
        Dim ptRef As String
        ptRef = Trim(CStr(wsThis.Cells(cfg, 1).Value))
        If ptRef = "" Then GoTo NextCfgNote

        Dim r As Long
        For r = PI_DATA_FIRST_ROW To piLR
            If Trim(CStr(wsPI.Cells(r, PI_COL_PTREF).Value)) = ptRef Then
                Dim fullNote As String
                fullNote = CStr(wsPI.Cells(r, PI_COL_NOTES).Value)
                If fullNote = "" Then GoTo FoundPI
                Dim lines() As String
                lines = Split(fullNote, Chr(10))
                Dim lastLine As String
                lastLine = Trim(lines(UBound(lines)))
                Dim pos As Long
                pos = InStr(LCase(lastLine), "notes: ")
                If pos > 0 Then
                    Dim fragment As String
                    fragment = Trim(Mid(lastLine, pos + 7))
                    ' Strip any trailing ZPI_SUFFIX (not a user note)
                    Dim zpPos As Long
                    zpPos = InStr(LCase(fragment), LCase(ZPI_SUFFIX))
                    If zpPos > 0 Then fragment = Left(fragment, zpPos - 1)
                    wsThis.Cells(cfg, TH_COL_NOTES).Value = fragment
                End If
FoundPI:
                Exit For
            End If
        Next r
NextCfgNote:
    Next cfg
End Sub

' ============================================================
' RebuildDataTable
'
' Clears the data table (row 46+) then repopulates every sensor
' column for timestamps in [StartDate, EndDate].
'
' Performance: pre-loads PD header + offset rows and all PD
' timestamps into module-level arrays (mBatch_*) so that
' PopulateOneSensor uses array lookups, not cell reads.
' Application.Calculation is left on Automatic throughout
' (switching to Manual and back was the main slowdown identified
'  in v1.62).
' ============================================================
Sub RebuildDataTable()
    Dim wsThis As Worksheet
    Dim wsPD   As Worksheet
    On Error GoTo Done
    Application.ScreenUpdating = False
    ' Note: EnableEvents is intentionally NOT disabled here.
    ' All writes are either to rows 45+ (outside the 9-43 config
    ' range checked by every handler) or are multi-cell Resize()
    ' calls that Worksheet_Change exits on immediately.
    ' Individual callers (clear, load, paste) guard their own
    ' config-area writes with EnableEvents=False as needed.

    Set wsThis = ThisWorkbook.Sheets("Total Head")
    Set wsPD   = ThisWorkbook.Sheets("Pressure Data")

    ' ── Date range ──────────────────────────────────────────
    Dim rawStart As Variant, rawEnd As Variant
    rawStart = wsThis.Cells(START_DATE_ROW, START_DATE_COL).Value
    rawEnd   = wsThis.Cells(END_DATE_ROW,   END_DATE_COL).Value
    If IsEmpty(rawStart) Or IsEmpty(rawEnd) Then GoTo Done
    Dim startD As Double, endD As Double
    startD = CDbl(rawStart)
    endD   = CDbl(rawEnd)
    If endD < startD Then GoTo Done

    ' ── Pre-load PD header + offset (read each exactly once) ─
    Dim pdLastCol As Long
    pdLastCol = wsPD.Cells(PD_HEADER_ROW, wsPD.Columns.Count).End(xlToLeft).Column
    ReDim mBatch_pdHdr(1 To pdLastCol)
    ReDim mBatch_pdOfs(1 To pdLastCol)
    Dim c As Long
    For c = 1 To pdLastCol
        mBatch_pdHdr(c) = CStr(wsPD.Cells(PD_HEADER_ROW, c).Value)
        If IsNumeric(wsPD.Cells(PD_OFFSET_ROW, c).Value) Then
            mBatch_pdOfs(c) = CDbl(wsPD.Cells(PD_OFFSET_ROW, c).Value)
        End If
    Next c
    mBatch_pdLastCol = pdLastCol

    ' ── Pre-load PD timestamp column ────────────────────────
    Dim pdLastRow As Long
    pdLastRow = wsPD.Cells(wsPD.Rows.Count, 1).End(xlUp).Row
    ReDim mBatch_pdTS(PD_FIRST_DATA_ROW To pdLastRow)
    Dim rr As Long
    For rr = PD_FIRST_DATA_ROW To pdLastRow
        Dim tv As Variant
        tv = wsPD.Cells(rr, 1).Value
        If Not IsEmpty(tv) Then mBatch_pdTS(rr) = CDbl(tv)
    Next rr
    mBatch_pdFirstDataRow = PD_FIRST_DATA_ROW
    mBatch_pdLastRow      = pdLastRow
    mBatch_Active         = True

    ' ── Pre-load PI Logger ID mapping (for FindPressureDataCol fallback) ─
    ' Loads PI Point Ref + Logger ID (col R) so that sensors whose PD
    ' column header is a Logger ID rather than a Point Ref are still found.
    mBatch_piCount = 0
    Dim wsPI As Worksheet
    On Error Resume Next
    Set wsPI = ThisWorkbook.Sheets("Point Index")
    On Error GoTo Done
    If Not wsPI Is Nothing Then
        Dim piLR As Long
        piLR = wsPI.Cells(wsPI.Rows.Count, PI_COL_PTREF).End(xlUp).Row
        If piLR >= PI_DATA_FIRST_ROW Then
            Dim piMax As Long
            piMax = piLR - PI_DATA_FIRST_ROW + 1
            ReDim mBatch_piRef(1 To piMax)
            ReDim mBatch_piLogger(1 To piMax)
            Dim piIdx As Long
            For piIdx = PI_DATA_FIRST_ROW To piLR
                Dim piRefStr As String
                piRefStr = Trim(CStr(wsPI.Cells(piIdx, PI_COL_PTREF).Value))
                If piRefStr <> "" Then
                    mBatch_piCount = mBatch_piCount + 1
                    mBatch_piRef(mBatch_piCount)    = piRefStr
                    mBatch_piLogger(mBatch_piCount) = Trim(CStr(wsPI.Cells(piIdx, PI_COL_LOGGER_ID).Value))
                End If
            Next piIdx
        End If
    End If

    ' ── Clear existing data area ─────────────────────────────
    Dim lastTHRow As Long
    lastTHRow = wsThis.Cells(wsThis.Rows.Count, 1).End(xlUp).Row
    If lastTHRow >= DATA_HDR_ROW Then
        wsThis.Range( _
            wsThis.Cells(DATA_HDR_ROW, 1), _
            wsThis.Cells(Application.Max(lastTHRow, DATA_HDR_ROW), CFG_LAST - 7 + 1) _
        ).ClearContents
    End If
    wsThis.Cells(DATA_HDR_ROW, 1).Value = "Timestamp"

    ' ── Build TH timestamp column from PD timestamps in range ─
    Dim thRow As Long
    thRow = DATA_FIRST_ROW
    ReDim mBatch_thTS(DATA_FIRST_ROW To DATA_FIRST_ROW + mBatch_pdLastRow - PD_FIRST_DATA_ROW)
    For rr = PD_FIRST_DATA_ROW To pdLastRow
        Dim ts As Double
        ts = mBatch_pdTS(rr)
        If ts >= startD And ts <= endD + 1 Then
            wsThis.Cells(thRow, 1).Value = wsPD.Cells(rr, 1).Value
            mBatch_thTS(thRow) = ts
            thRow = thRow + 1
        End If
    Next rr
    mBatch_thLastRow = thRow - 1
    If thRow <= DATA_FIRST_ROW Then GoTo Done   ' no data in range

    ' ── Populate each sensor column ─────────────────────────
    Dim cfg As Long
    Dim sensorsTotal As Long
    Dim sensorsFound As Long
    For cfg = CFG_FIRST To CFG_LAST
        Dim ptRef As String
        ptRef = Trim(CStr(wsThis.Cells(cfg, 1).Value))
        If ptRef = "" Then GoTo NextSensor

        sensorsTotal = sensorsTotal + 1
        Dim dataCol As Long
        dataCol = cfg - 7   ' row 9 → col 2 (B), row 10 → col 3 (C), …
        wsThis.Cells(DATA_HDR_ROW, dataCol).Value = ptRef
        If FindPressureDataCol(ptRef) > 0 Then sensorsFound = sensorsFound + 1
        Call PopulateOneSensor(wsThis, wsPD, cfg, dataCol, startD, endD)
NextSensor:
    Next cfg

    ' ── Status bar + alert: tell user how many sensors were matched ──
    If sensorsTotal > 0 Then
        If sensorsFound = 0 Then
            ' ── ALL sensors missed: show a diagnostic MsgBox so the
            '    user cannot miss the mismatch. ──────────────────────
            Dim missedList  As String
            Dim pdHdrList   As String
            Dim mm As Long
            For mm = CFG_FIRST To CFG_LAST
                Dim mRef As String
                mRef = Trim(CStr(wsThis.Cells(mm, 1).Value))
                If mRef <> "" Then
                    missedList = missedList & "  • " & mRef & vbNewLine
                End If
            Next mm
            ' Build a short list of what IS in PD row 7
            Dim pdCount As Long
            pdCount = 0
            Dim hh As Long
            For hh = 2 To mBatch_pdLastCol
                If mBatch_pdHdr(hh) <> "" Then
                    pdHdrList = pdHdrList & "  " & mBatch_pdHdr(hh)
                    pdCount = pdCount + 1
                    If pdCount = 8 Then
                        pdHdrList = pdHdrList & " …"
                        Exit For
                    End If
                End If
            Next hh
            If pdHdrList = "" Then pdHdrList = "  (none found in row 7)"
            MsgBox "No sensor data was populated." & vbNewLine & vbNewLine & _
                   "Your configured Point Refs:" & vbNewLine & missedList & vbNewLine & _
                   "Headers found in Pressure Data row 7:" & vbNewLine & pdHdrList & vbNewLine & vbNewLine & _
                   "Check that the Point Ref names match the Pressure Data column" & vbNewLine & _
                   "headers exactly (names are compared case-insensitively).", _
                   vbExclamation, "Total Head – Sensor Mismatch"
            Application.StatusBar = "Total Head: 0/" & sensorsTotal & _
                " sensor(s) matched. Check Pressure Data headers."
        ElseIf sensorsFound < sensorsTotal Then
            Application.StatusBar = "Total Head: " & sensorsFound & "/" & sensorsTotal & _
                " sensor(s) matched in Pressure Data."
        Else
            Application.StatusBar = "Total Head: " & sensorsFound & "/" & sensorsTotal & _
                " sensor(s) populated."
        End If
    End If

Done:
    mBatch_Active   = False
    mBatch_piCount  = 0
    Application.ScreenUpdating = True
    On Error GoTo 0
End Sub

' ============================================================
' PopulateOneSensor
' Fills one sensor data column using the mBatch_* cache.
' Two-pointer (cursor) algorithm: O(n + m) timestamp matching.
'
' Output formula:  Total Head = pressure + offset + Final Z
'   where pressure = raw value from Pressure Data sheet,
'         offset   = TH col B (calibration zero correction),
'         Final Z  = TH col D (elevation above datum).
' If Final Z is 0 or empty, it contributes 0 (no elevation component).
' ============================================================
Sub PopulateOneSensor(wsThis As Worksheet, wsPD As Worksheet, _
                      cfgRow As Long, dataCol As Long, _
                      startD As Double, endD As Double)
    Dim ptRef As String
    ptRef = Trim(CStr(wsThis.Cells(cfgRow, 1).Value))
    If ptRef = "" Then Exit Sub

    Dim pdCol As Long
    pdCol = FindPressureDataCol(ptRef)
    If pdCol = 0 Then Exit Sub

    Dim offset As Double
    Dim thOfs As Variant
    thOfs = wsThis.Cells(cfgRow, TH_COL_OFFSET).Value
    If IsNumeric(thOfs) Then offset = CDbl(thOfs)

    ' Final Z elevation component: added to pressure + offset for Total Head
    Dim finalZ As Double
    finalZ = 0  ' default: no elevation component if Z is missing or zero
    Dim thZRaw As Variant
    thZRaw = wsThis.Cells(cfgRow, TH_COL_FINAL_Z).Value
    If IsNumeric(thZRaw) Then
        Dim thZd As Double : thZd = CDbl(thZRaw)
        If thZd <> 0 Then finalZ = thZd
    End If

    ' Pre-load PD data column in one bulk read (1 crossing vs ~3000)
    Dim pdDataRaw As Variant
    If mBatch_pdLastRow > mBatch_pdFirstDataRow Then
        pdDataRaw = wsPD.Range( _
            wsPD.Cells(mBatch_pdFirstDataRow, pdCol), _
            wsPD.Cells(mBatch_pdLastRow, pdCol)).Value
    Else
        ' Single row (or empty) — build minimal 2D array manually
        ReDim pdDataRaw(1 To 1, 1 To 1)
        If mBatch_pdLastRow = mBatch_pdFirstDataRow Then
            pdDataRaw(1, 1) = wsPD.Cells(mBatch_pdFirstDataRow, pdCol).Value
        End If
    End If

    ' Build output array (written in one block at the end)
    Dim thLastRow As Long
    If mBatch_Active Then
        thLastRow = mBatch_thLastRow
    Else
        thLastRow = wsThis.Cells(wsThis.Rows.Count, 1).End(xlUp).Row
    End If
    If thLastRow < DATA_FIRST_ROW Then Exit Sub

    Dim outCount As Long
    outCount = thLastRow - DATA_FIRST_ROW + 1
    Dim outArr() As Variant
    ReDim outArr(1 To outCount, 1 To 1)

    Dim jj As Long   ' cursor into PD (advances monotonically)
    jj = mBatch_pdFirstDataRow

    Dim thRow As Long
    For thRow = DATA_FIRST_ROW To thLastRow
        Dim thTSd As Double
        If mBatch_Active Then
            thTSd = mBatch_thTS(thRow)
            If thTSd = 0 Then GoTo NextThRow
        Else
            Dim thTSv As Variant
            thTSv = wsThis.Cells(thRow, 1).Value
            If IsEmpty(thTSv) Then GoTo NextThRow
            thTSd = CDbl(thTSv)
        End If

        ' Advance PD cursor to matching or later timestamp
        Do While jj <= mBatch_pdLastRow And mBatch_pdTS(jj) < thTSd - TIMESTAMP_TOLERANCE
            jj = jj + 1
        Loop

        If jj <= mBatch_pdLastRow And Abs(mBatch_pdTS(jj) - thTSd) < TIMESTAMP_TOLERANCE Then
            Dim v As Variant
            v = pdDataRaw(jj - mBatch_pdFirstDataRow + 1, 1)
            If IsNumeric(v) And CDbl(v) <> PD_MISSING_VALUE Then
                outArr(thRow - DATA_FIRST_ROW + 1, 1) = CDbl(v) + offset + finalZ
            End If
        End If
NextThRow:
    Next thRow

    wsThis.Cells(DATA_FIRST_ROW, dataCol).Resize(outCount, 1).Value = outArr
End Sub

' ============================================================
' LoadTotalHeadByName  (RECALL)
'
' OPTIMISED: reads each source tab exactly once.
'   1. Total Head Archive  – read sensor list (one row)
'   2. Pressure Data       – pre-load header row + offset row
'   3. Point Index         – pre-load all Point Ref + Z rows
' Then restore config with array lookups (no per-sensor reads).
' Finally, call RebuildDataTable which also uses array caching.
'
' Cost is the same as typing a single Point Ref into the config
' (which scans PD headers once + PI once), just covering all
' sensors from the 3 source tabs instead of 2.
' ============================================================
Sub LoadTotalHeadByName(zoneName As String)
    Dim wsThis As Worksheet
    Dim wsArch As Worksheet
    Dim wsPD   As Worksheet
    Dim wsPI   As Worksheet
    On Error GoTo ErrOut

    Set wsThis = ThisWorkbook.Sheets("Total Head")
    Set wsArch = ThisWorkbook.Sheets("Total Head Archive")
    Set wsPD   = ThisWorkbook.Sheets("Pressure Data")
    Set wsPI   = ThisWorkbook.Sheets("Point Index")

    Application.StatusBar = "Loading: " & zoneName & " ..."
    Application.ScreenUpdating = False
    Application.EnableEvents   = False

    ' ── TAB 1: Find archive row ───────────────────────────
    Dim lastArchRow As Long
    lastArchRow = wsArch.Cells(wsArch.Rows.Count, ARCH_COL_ZONE).End(xlUp).Row

    Dim archRow As Long
    archRow = 0
    Dim r As Long
    For r = 2 To lastArchRow
        If Trim(CStr(wsArch.Cells(r, ARCH_COL_ZONE).Value)) = Trim(zoneName) Then
            archRow = r
            Exit For
        End If
    Next r

    If archRow = 0 Then
        Application.EnableEvents   = True
        Application.ScreenUpdating = True
        MsgBox "Zone '" & zoneName & "' not found in archive.", vbExclamation
        Exit Sub
    End If

    ' Read sensor list from archive row – one bulk read (1 crossing vs 35)
    Dim archSensors(1 To MAX_SENSORS) As String
    Dim archRaw As Variant
    archRaw = wsArch.Range( _
        wsArch.Cells(archRow, ARCH_COL_SENSOR1), _
        wsArch.Cells(archRow, ARCH_COL_SENSOR1 + MAX_SENSORS - 1)).Value
    Dim s As Long
    For s = 1 To MAX_SENSORS
        Dim av As Variant
        av = archRaw(1, s)
        If Not IsEmpty(av) And CStr(av) <> "" Then archSensors(s) = Trim(CStr(av))
    Next s

    ' ── TAB 2: Pre-load Pressure Data in one pass ─────────
    Dim pdLastCol As Long
    pdLastCol = wsPD.Cells(PD_HEADER_ROW, wsPD.Columns.Count).End(xlToLeft).Column

    Dim pdHdr() As String
    Dim pdOfs() As Double
    ReDim pdHdr(1 To pdLastCol)
    ReDim pdOfs(1 To pdLastCol)

    Dim c As Long
    For c = 1 To pdLastCol
        pdHdr(c) = CStr(wsPD.Cells(PD_HEADER_ROW, c).Value)
        If IsNumeric(wsPD.Cells(PD_OFFSET_ROW, c).Value) Then
            pdOfs(c) = CDbl(wsPD.Cells(PD_OFFSET_ROW, c).Value)
        End If
    Next c

    ' ── TAB 3: Pre-load Point Index in one pass ───────────
    Dim piLastRow As Long
    piLastRow = wsPI.Cells(wsPI.Rows.Count, PI_COL_PTREF).End(xlUp).Row

    Dim piRefs As Variant
    Dim piZArr As Variant
    Dim piGpsArr As Variant
    Dim piDipArr As Variant
    Dim piDumpyArr As Variant
    ReDim piRefs(1 To piLastRow - PI_DATA_FIRST_ROW + 1)
    ReDim piZArr(1 To piLastRow - PI_DATA_FIRST_ROW + 1)
    ReDim piGpsArr(1 To piLastRow - PI_DATA_FIRST_ROW + 1)
    ReDim piDipArr(1 To piLastRow - PI_DATA_FIRST_ROW + 1)
    ReDim piDumpyArr(1 To piLastRow - PI_DATA_FIRST_ROW + 1)

    Dim piCount As Long
    For r = PI_DATA_FIRST_ROW To piLastRow
        Dim pref As String
        pref = Trim(CStr(wsPI.Cells(r, PI_COL_PTREF).Value))
        If pref <> "" Then
            piCount = piCount + 1
            piRefs(piCount)    = pref
            piZArr(piCount)    = wsPI.Cells(r, PI_COL_FINAL_Z).Value
            piGpsArr(piCount)  = wsPI.Cells(r, PI_COL_GPS).Value
            piDipArr(piCount)  = wsPI.Cells(r, PI_COL_DIP).Value
            piDumpyArr(piCount) = wsPI.Cells(r, PI_COL_DUMPY).Value
        End If
    Next r

    ' ── Restore metadata ─────────────────────────────────
    wsThis.Cells(ZONE_ROW,  ZONE_COL).Value    = wsArch.Cells(archRow, ARCH_COL_ZONE).Value
    wsThis.Cells(ZONE_ROW,  DRV_HEAD_COL).Value = wsArch.Cells(archRow, ARCH_COL_DRV_HEAD).Value
    wsThis.Cells(NOTES_ROW, NOTES_COL).Value   = wsArch.Cells(archRow, ARCH_COL_NOTES).Value

    ' ── Clear config rows ─────────────────────────────────
    Dim cfg As Long
    For cfg = CFG_FIRST To CFG_LAST
        wsThis.Cells(cfg, 1).Value = ""  ' Point Ref
        wsThis.Cells(cfg, TH_COL_OFFSET).Value = ""  ' Offset
        wsThis.Cells(cfg, TH_COL_GPS).Value   = ""   ' GPS
        wsThis.Cells(cfg, TH_COL_FINAL_Z).Value = "" ' Z
        wsThis.Cells(cfg, TH_COL_DIP).Value   = ""   ' Dip
        wsThis.Cells(cfg, TH_COL_DUMPY).Value = ""   ' Dumpy
        wsThis.Cells(cfg, TH_COL_NOTES).Value = ""   ' Notes
    Next cfg

    ' ── Restore sensors using pre-loaded arrays ───────────
    ' For each sensor in the archive: array lookups only —
    ' no per-sensor sheet reads.  This is the key optimisation:
    ' total work is O(pdCols + piRows + 35*max(pdCols,piRows)/2)
    ' vs the old O(35 * (pdCols + piRows)).
    For s = 1 To MAX_SENSORS
        Dim ptRef As String
        ptRef = archSensors(s)
        If ptRef = "" Then GoTo NextSensorLoad

        cfg = CFG_FIRST + s - 1
        wsThis.Cells(cfg, 1).Value = ptRef

        ' Offset: scan pre-loaded PD header array
        For c = 2 To pdLastCol
            If LCase(pdHdr(c)) = LCase(ptRef) Then
                wsThis.Cells(cfg, TH_COL_OFFSET).Value = pdOfs(c)
                Exit For
            End If
        Next c

        ' Z / GPS / Dip / Dumpy: scan pre-loaded PI array
        Dim pi As Long
        For pi = 1 To piCount
            If piRefs(pi) = ptRef Then
                Dim zv As Variant
                zv = piZArr(pi)
                If Not IsEmpty(zv) And IsNumeric(zv) Then
                    wsThis.Cells(cfg, TH_COL_FINAL_Z).Value = CDbl(zv)
                End If
                Dim gpsV As Variant : gpsV = piGpsArr(pi)
                Dim dipV As Variant : dipV = piDipArr(pi)
                Dim dumpyV As Variant : dumpyV = piDumpyArr(pi)
                If Not IsEmpty(gpsV)   And IsNumeric(gpsV)   Then wsThis.Cells(cfg, TH_COL_GPS).Value   = CDbl(gpsV)
                If Not IsEmpty(dipV)   And IsNumeric(dipV)   Then wsThis.Cells(cfg, TH_COL_DIP).Value   = CDbl(dipV)
                If Not IsEmpty(dumpyV) And IsNumeric(dumpyV) Then wsThis.Cells(cfg, TH_COL_DUMPY).Value = CDbl(dumpyV)
                Exit For
            End If
        Next pi
NextSensorLoad:
    Next s

    ' ── Safety pass: clear B-F for any row still without a Point Ref ─
    ' Ensures stale GPS/Z/Dip/Dumpy/Offset data from a previous
    ' session is never left visible when the new zone has fewer sensors.
    For cfg = CFG_FIRST To CFG_LAST
        If Trim(CStr(wsThis.Cells(cfg, 1).Value)) = "" Then
            wsThis.Cells(cfg, TH_COL_OFFSET).Value  = ""
            wsThis.Cells(cfg, TH_COL_GPS).Value     = ""
            wsThis.Cells(cfg, TH_COL_FINAL_Z).Value = ""
            wsThis.Cells(cfg, TH_COL_DIP).Value     = ""
            wsThis.Cells(cfg, TH_COL_DUMPY).Value   = ""
            wsThis.Cells(cfg, TH_COL_NOTES).Value   = ""
        End If
    Next cfg

    Application.EnableEvents = True

    ' ── Rebuild data table (uses its own mBatch_* cache) ──
    Call RebuildDataTable
    Call CreateTotalHeadChart

    Application.StatusBar = "Loaded: " & zoneName
    Exit Sub

ErrOut:
    Application.EnableEvents   = True
    Application.ScreenUpdating = True
    Application.StatusBar      = False
    MsgBox "LoadTotalHeadByName error: " & Err.Description, vbCritical
End Sub

' ============================================================
' CREATE / VERIFY TOTAL HEAD CHART
' ============================================================
Sub CreateTotalHeadChart()
    Dim wsThis As Worksheet
    Set wsThis = ThisWorkbook.Sheets("Total Head")

    Const CHART_NAME As String  = "TotalHeadChart"
    Const ANCHOR_COL As Long    = 15        ' Column O
    Const ANCHOR_ROW As Long    = 3
    Const CHART_W_CM As Double  = 37.17
    Const CHART_H_CM As Double  = 22.28

    Dim cht As ChartObject
    On Error Resume Next
    Set cht = wsThis.ChartObjects(CHART_NAME)
    On Error GoTo 0

    If cht Is Nothing Then
        Dim co As ChartObject
        For Each co In wsThis.ChartObjects
            co.Delete
        Next co
        Set cht = wsThis.ChartObjects.Add( _
            wsThis.Cells(ANCHOR_ROW, ANCHOR_COL).Left, _
            wsThis.Cells(ANCHOR_ROW, ANCHOR_COL).Top, _
            Application.CentimetersToPoints(CHART_W_CM), _
            Application.CentimetersToPoints(CHART_H_CM))
        cht.Name = CHART_NAME
        cht.Chart.ChartType = xlXYScatterSmooth
    End If

    Call RefreshTotalHeadChart
End Sub

' ============================================================
' REFRESH TOTAL HEAD CHART
' ============================================================
Sub RefreshTotalHeadChart()
    Dim wsThis As Worksheet
    Set wsThis = ThisWorkbook.Sheets("Total Head")

    Dim cht As ChartObject
    On Error Resume Next
    Set cht = wsThis.ChartObjects("TotalHeadChart")
    On Error GoTo 0
    If cht Is Nothing Then Exit Sub

    Dim ch As Chart
    Set ch = cht.Chart

    Do While ch.SeriesCollection.Count > 0
        ch.SeriesCollection(1).Delete
    Loop

    Dim lastTHRow As Long
    lastTHRow = wsThis.Cells(wsThis.Rows.Count, 1).End(xlUp).Row
    If lastTHRow < DATA_FIRST_ROW Then Exit Sub

    ' Add one series per active sensor
    Dim cfg As Long
    For cfg = CFG_FIRST To CFG_LAST
        Dim ptRef As String
        ptRef = Trim(CStr(wsThis.Cells(cfg, 1).Value))
        If ptRef = "" Then GoTo NextChSer

        Dim dataCol As Long
        dataCol = cfg - 7

        ' Only add if the column has at least one value
        Dim hasData As Boolean
        Dim rr As Long
        For rr = DATA_FIRST_ROW To Application.Min(DATA_FIRST_ROW + 10, lastTHRow)
            If Not IsEmpty(wsThis.Cells(rr, dataCol).Value) Then
                hasData = True
                Exit For
            End If
        Next rr
        If Not hasData Then GoTo NextChSer

        Dim ser As Series
        Set ser = ch.SeriesCollection.NewSeries
        ser.Name       = ptRef
        ser.XValues    = wsThis.Range(wsThis.Cells(DATA_FIRST_ROW, 1), _
                                      wsThis.Cells(lastTHRow, 1))
        ser.Values     = wsThis.Range(wsThis.Cells(DATA_FIRST_ROW, dataCol), _
                                      wsThis.Cells(lastTHRow, dataCol))
        ser.MarkerStyle = xlMarkerStyleNone
NextChSer:
    Next cfg

    ' Chart title
    Dim titleStr As String
    titleStr = Trim(CStr(wsThis.Cells(ZONE_ROW, ZONE_COL).Value))
    On Error Resume Next
    If titleStr <> "" Then
        ch.HasTitle = True
        ch.ChartTitle.Text = titleStr
    End If
    On Error GoTo 0

    ' X-axis scale
    Dim startD As Variant, endD As Variant
    startD = wsThis.Cells(START_DATE_ROW, START_DATE_COL).Value
    endD   = wsThis.Cells(END_DATE_ROW,   END_DATE_COL).Value

    On Error Resume Next
    Dim ax As Axis
    Set ax = ch.Axes(xlCategory)

    If Not IsEmpty(startD) And IsNumeric(startD) Then
        ax.MinimumScale = Int(CDbl(startD))
    Else
        ax.MinimumScaleIsAuto = True
    End If
    If Not IsEmpty(endD) And IsNumeric(endD) Then
        ax.MaximumScale = Int(CDbl(endD)) + 1
    Else
        ax.MaximumScaleIsAuto = True
    End If

    ' Tick intervals based on span
    If Not IsEmpty(startD) And Not IsEmpty(endD) Then
        Dim span As Double
        span = CDbl(endD) - CDbl(startD)
        If span <= 1 Then
            ax.MajorUnit = 1 / 24 : ax.MinorUnit = 1 / 24
        ElseIf span <= 3 Then
            ax.MajorUnit = 6 / 24 : ax.MinorUnit = 1 / 24
        ElseIf span <= 7 Then
            ax.MajorUnit = 1      : ax.MinorUnit = 6 / 24
        ElseIf span <= 31 Then
            ax.MajorUnit = 7      : ax.MinorUnit = 1
        Else
            ax.MajorUnit = 14     : ax.MinorUnit = 7
        End If
        ax.MajorUnitScale = xlDays
        ax.MinorUnitScale = xlDays
    End If
    On Error GoTo 0

    ch.Axes(xlValue).MinimumScaleIsAuto = True
    ch.Axes(xlValue).MaximumScaleIsAuto = True
End Sub

' ============================================================
' FindInstallRetrievalDates
' Scans PI row 2 for "INSTALL"/"RETRIEVAL" group headers,
' then row 3 for "date" sub-headers (handles "retrival" typo).
' ============================================================
Function FindInstallRetrievalDates(ptRef As String, _
                                    ByRef installDate As Variant, _
                                    ByRef retrievalDate As Variant) As Boolean
    Dim wsPI As Worksheet
    Set wsPI = ThisWorkbook.Sheets("Point Index")
    installDate   = Empty
    retrievalDate = Empty
    FindInstallRetrievalDates = False

    Dim lastCol As Long
    lastCol = wsPI.Cells(2, wsPI.Columns.Count).End(xlToLeft).Column

    Dim installCol As Long, retrievalCol As Long
    installCol   = 0
    retrievalCol = 0

    Dim c As Long, c2 As Long
    For c = 1 To lastCol
        Dim h2 As String
        h2 = UCase(Trim(CStr(wsPI.Cells(2, c).Value)))
        If h2 = "INSTALL" And installCol = 0 Then
            For c2 = c To Application.Min(c + 5, lastCol)
                If InStr(LCase(CStr(wsPI.Cells(3, c2).Value)), "date") > 0 Then
                    installCol = c2 : Exit For
                End If
            Next c2
        End If
        If (h2 = "RETRIEVAL" Or h2 = "RETRIVAL") And retrievalCol = 0 Then
            For c2 = c To Application.Min(c + 5, lastCol)
                If InStr(LCase(CStr(wsPI.Cells(3, c2).Value)), "date") > 0 Then
                    retrievalCol = c2 : Exit For
                End If
            Next c2
        End If
    Next c

    If installCol = 0 Then Exit Function

    Dim lr As Long
    lr = wsPI.Cells(wsPI.Rows.Count, PI_COL_PTREF).End(xlUp).Row
    For c = PI_DATA_FIRST_ROW To lr
        If Trim(CStr(wsPI.Cells(c, PI_COL_PTREF).Value)) = ptRef Then
            installDate = wsPI.Cells(c, installCol).Value
            If retrievalCol > 0 Then retrievalDate = wsPI.Cells(c, retrievalCol).Value
            FindInstallRetrievalDates = True
            Exit Function
        End If
    Next c
End Function

' ============================================================
' ExportTotalHeadPRN
' Writes a tab-delimited text file to the path in Front Page B6.
' ============================================================
Sub ExportTotalHeadPRN()
    Dim wsThis As Worksheet
    Set wsThis = ThisWorkbook.Sheets("Total Head")

    Dim exportPath As String
    On Error Resume Next
    exportPath = CStr(ThisWorkbook.Sheets("Front Page").Cells(6, 2).Value)
    On Error GoTo 0
    If exportPath = "" Then
        MsgBox "Set an export path in Front Page cell B6.", vbExclamation
        Exit Sub
    End If
    If Right(exportPath, 1) <> "\" Then exportPath = exportPath & "\"

    Dim zoneName As String
    zoneName = Trim(CStr(wsThis.Cells(ZONE_ROW, ZONE_COL).Value))
    If zoneName = "" Then zoneName = "TotalHead"

    Dim filename As String
    filename = exportPath & zoneName & "_" & Format(Now(), "yyyymmdd_hhnnss") & ".prn"

    Dim fn As Integer
    fn = FreeFile
    Open filename For Output As #fn

    ' Header row
    Print #fn, "Timestamp" & vbTab;
    Dim cfg As Long
    For cfg = CFG_FIRST To CFG_LAST
        If Trim(CStr(wsThis.Cells(cfg, 1).Value)) <> "" Then
            Print #fn, wsThis.Cells(cfg, 1).Value & vbTab;
        End If
    Next cfg
    Print #fn, ""

    ' Data rows
    Dim lastRow As Long
    lastRow = wsThis.Cells(wsThis.Rows.Count, 1).End(xlUp).Row
    Dim rr As Long
    For rr = DATA_FIRST_ROW To lastRow
        Dim tsv As Variant
        tsv = wsThis.Cells(rr, 1).Value
        If IsEmpty(tsv) Then GoTo NextPRNRow
        Print #fn, Format(CDate(tsv), DATE_FMT) & vbTab;
        For cfg = CFG_FIRST To CFG_LAST
            If Trim(CStr(wsThis.Cells(cfg, 1).Value)) <> "" Then
                Dim v As Variant
                v = wsThis.Cells(rr, cfg - 7).Value
                Print #fn, IIf(IsEmpty(v), "", CStr(CDbl(v))) & vbTab;
            End If
        Next cfg
        Print #fn, ""
NextPRNRow:
    Next rr

    Close #fn
    MsgBox "Exported to: " & filename, vbInformation
End Sub
