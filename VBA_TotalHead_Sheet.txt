' Total Head (New) Worksheet — Event Handlers  (v1.57)
' How to install:
'   1. Alt+F11 to open the VBA editor.
'   2. In the Project pane, expand "Microsoft Excel Objects".
'   3. Double-click "Total Head".
'   4. Paste this entire block, replacing any existing code.
' ============================================================
'
' ── Layout reference ────────────────────────────────────────────────────────
'   B3  = "Zone Name:"    label      D3  = zone name / chart title input  (merged D3:I4)
'   B4  = "System:"       label      C4  = system name input
'   B5  = "Start Date:"   label      D5  = start date input  (merged D5:E5)
'   B6  = "End Date:"     label      D6  = end date input    (merged D6:E6)
'   J3  = "Driving Head:" label      L3  = driving head sensor  (merged L3:N4)
'   I5  = "SAVE"  (click-button, merged I5:I6)
'   K5  = "Load"    label  (not a button — use L5:N6 dropdown to load)
'   L5  = archive load dropdown  (merged L5:N6; data-validation list of saved
'         Zone Names; displays current Zone Name (D3) when no session loaded;
'         selecting a name loads that session without the InputBox)
'   J8  = "Zone Notes"    label
'   J9  = zone notes input  (merged J9:N43)
'   Row 8  : table header
'            A=Point Ref | B=Offset | C=(unused) | D=Z(m) | E=(reserved) | F=PRN | G=Point Notes
'   Rows 9-43 : sensor config rows (35 rows, col A = yellow Point Ref input)
'   Row 44 : spacer
'   Row 45 : data table headers  (A=Timestamp, B..AJ = one col per sensor)
'   Row 46+: data rows            (A=date+time serial, B..AJ = pressure values)
'   Chart  : "TotalHeadChart" anchored at O3, created / refreshed by VBA
'   Export path: read from Front Page B6 (not stored on this sheet)
' ────────────────────────────────────────────────────────────────────────────

' Module-level arrays: snapshot state per config row.
'   mPrevRefByRow : Point Ref as it was before the user edited/cleared it
'   mOrigZByRow   : Z(m) as filled from Point Index when the ref was entered —
'                   used so RecordSensorNote can include the original PI Z even
'                   after WriteElevationToPointIndex has already overwritten PI.
Private mPrevRefByRow(9 To 43) As String
Private mOrigZByRow(9 To 43)   As Variant

Private Sub Worksheet_Activate()
    ' Initialise sheet labels and ensure chart exists and is linked to current data.
    ' Pattern mirrors the Plots sheet: only write to cells (RebuildDataTable) when
    ' the chart must be created from scratch.  On subsequent activations we only
    ' call RefreshTotalHeadChart, which is a pure chart-object operation and never
    ' writes to worksheet cells — so Application.CutCopyMode is left intact and
    ' a Ctrl+V paste the user has queued up is not silently discarded.
    Application.EnableEvents = False
    On Error Resume Next
    InitTotalHeadSheet Me
    PopulateAllElevationsTH Me   ' fill Z(m) from Point Index for any row with a ref but no Z

    Dim co As ChartObject
    Dim chartOk As Boolean: chartOk = False
    For Each co In Me.ChartObjects
        If co.Name = "TotalHeadChart" Then chartOk = True: Exit For
    Next co

    If Not chartOk Then
        ' First visit or chart was deleted: populate the data table first so the
        ' chart has series to display immediately on creation (same logic as v1.50,
        ' but only runs when the chart is actually absent — not on every activate).
        For Each co In Me.ChartObjects: co.Delete: Next co
        RebuildDataTable Me
        CreateTotalHeadChart Me   ' also calls RefreshTotalHeadChart internally
    ElseIf Me.Cells(46, 1).Value = "" Then
        ' Chart exists but data table is empty (workbook just opened / data cleared):
        ' rebuild now.  Only fires when A46 is genuinely empty, so the clipboard is
        ' not disturbed on normal tab re-visits when data is already present.
        RebuildDataTable Me
        RefreshTotalHeadChart Me
    Else
        ' Chart exists and data table is populated — repaint chart only, no cell writes.
        RefreshTotalHeadChart Me
    End If

    Application.EnableEvents = True
    On Error GoTo 0
End Sub

Private Sub Worksheet_Change(ByVal Target As Range)

    Const CONF_START  As Long = 9    ' first sensor config row
    Const CONF_END    As Long = 43   ' last sensor config row
    Const CONF_REF    As Long = 1    ' col A: Point Ref
    Const CONF_OFF    As Long = 2    ' col B: Offset
    Const CONF_ELEV   As Long = 4    ' col D: Z (m)
    Const CONF_NOTES  As Long = 7    ' col G: Point Notes

    Application.EnableEvents = False
    On Error Resume Next

    ' ── L5 = archive load dropdown — load the selected session ───────────────
    If Not Intersect(Target, Me.Range("L5")) Is Nothing Then
        Dim selZone As String
        selZone = Trim(CStr(Me.Range("L5").Value))
        If selZone <> "" And selZone <> Trim(CStr(Me.Cells(3, 4).Value)) Then
            LoadTotalHeadByName selZone
        End If
        GoTo Done
    End If

    ' ── D3 (Zone Name / Chart Title) changed → update chart title and L5 ─────
    If Not Intersect(Target, Me.Range("D3")) Is Nothing Then
        Dim newZone As String: newZone = Trim(CStr(Me.Cells(3, 4).Value))
        Dim coTitle As ChartObject
        For Each coTitle In Me.ChartObjects
            If coTitle.Name = "TotalHeadChart" Then
                coTitle.Chart.ChartTitle.Text = newZone
                Exit For
            End If
        Next coTitle
        ' Keep L5 in sync: always reflect the current Zone Name for new sessions
        Me.Range("L5").Value = IIf(newZone <> "", newZone, "Drop Down to load saved selections")
        GoTo Done
    End If

    ' ── D5 or D6 (Start / End Date) changed → full rebuild ───────────────────
    If Not Intersect(Target, Me.Range("D5:D6")) Is Nothing Then
        RebuildDataTable Me
        RefreshTotalHeadChart Me
        GoTo Done
    End If

    ' ── Sensor config changed but data table has no timestamps yet ───────────
    ' When the user pastes Point Refs (or edits Offset/Z) before a rebuild has
    ' run, PopulateOneSensor exits immediately because A46 is empty.  Fix: run
    ' a full rebuild first so timestamps are in place, then let the per-column
    ' handlers below update individual sensor columns as normal.
    ' Row 46 = TH_DATA_ROW (first data row), defined as a Private Const in Module3.
    If Me.Cells(46, 1).Value = "" Then
        If Not Intersect(Target, Me.Range("A9:D43")) Is Nothing Then
            RebuildDataTable Me
        End If
    End If

    ' ── Sensor config rows changed — handle per-column ────────────────────────
    Dim pCell As Range
    Dim needRefresh As Boolean: needRefresh = False

    For Each pCell In Target
        If pCell.Row >= CONF_START And pCell.Row <= CONF_END Then
            Select Case pCell.Column

                Case CONF_REF
                    ' Point Ref changed: refresh elevation, data column and header
                    Dim hdr As String
                    hdr = Trim(CStr(Me.Cells(pCell.Row, CONF_REF).Value))
                    Me.Cells(45, TH_DataCol(pCell.Row)).Value = hdr
                    If hdr = "" Then
                        ' ── Save note FIRST (before PopulateElevationTH clears col D) ─
                        ' If RecordSensorNote runs after PopulateElevationTH, col D is
                        ' already Empty and IsNumeric(Empty)=True→CDbl(Empty)=0 → "Z: 0."
                        ' Also pass the PI Z snapshot so the note can record the original.
                        RecordSensorNote Me, pCell.Row, mPrevRefByRow(pCell.Row), mOrigZByRow(pCell.Row)
                        PopulateElevationTH Me, pCell.Row   ' clears col D (ref is "")
                        Me.Cells(pCell.Row, CONF_OFF).Value = 0
                        Me.Cells(pCell.Row, CONF_NOTES).Value = ""   ' Value="" is more robust than ClearContents under On Error Resume Next
                        mPrevRefByRow(pCell.Row) = ""
                        mOrigZByRow(pCell.Row)   = Empty   ' clear snapshot
                    Else
                        ' Fill elevation from Point Index, then restore saved Offset and Notes.
                        ' Snapshot col D immediately after PopulateElevationTH fills it so we
                        ' retain the original PI Z even if the user later overrides col D.
                        PopulateElevationTH Me, pCell.Row
                        mOrigZByRow(pCell.Row) = Me.Cells(pCell.Row, CONF_ELEV).Value
                        RestorePointNotesTH Me, pCell.Row
                        mPrevRefByRow(pCell.Row) = hdr
                    End If
                    PopulateOneSensor Me, pCell.Row
                    needRefresh = True

                Case CONF_OFF, CONF_ELEV
                    ' Offset / Z(m) changed: sync offset to Pressure Data and/or
                    ' write Z back to Point Index, then repopulate data column
                    If pCell.Column = CONF_OFF  Then SyncOffsetToPressureData Me, pCell.Row
                    If pCell.Column = CONF_ELEV Then WriteElevationToPointIndex Me, pCell.Row
                    PopulateOneSensor Me, pCell.Row
                    needRefresh = True

                Case CONF_NOTES
                    ' Point Notes changed — no automatic Point Index write;
                    ' the note is saved to Point Index when the ref is cleared.

            End Select
        End If
    Next pCell

    If needRefresh Then RefreshTotalHeadChart Me

Done:
    Application.EnableEvents = True
    On Error GoTo 0
End Sub

Private Sub Worksheet_SelectionChange(ByVal Target As Range)

    Const CONF_START  As Long = 9
    Const CONF_END    As Long = 43
    Const CONF_PRN    As Long = 6    ' col F: PRN export button

    ' Allow merged-cell ranges through the guard
    If Target.Count > 1 Then
        If Not Target.MergeCells Then Exit Sub
    End If

    Dim cell As Range
    Set cell = Target.Cells(1, 1)

    ' ── Capture Point Ref before user edits it ───────────────────────────────
    ' Worksheet_Change fires after the cell is already changed, so we snapshot
    ' the current value here while it is still intact.
    If cell.Column = 1 And cell.Row >= CONF_START And cell.Row <= CONF_END Then
        mPrevRefByRow(cell.Row) = Trim(CStr(Me.Cells(cell.Row, 1).Value))
    End If

    ' ── I5 = SAVE button (merged I5:I6) ──────────────────────────────────────
    If cell.Row = 5 And cell.Column = 9 Then
        Application.EnableEvents = False
        cell.Offset(0, -1).Select
        Application.EnableEvents = True
        Call SaveTotalHead
        Exit Sub
    End If

    ' ── Col F rows 9-43 = PRN export — only when Point Ref present ───────────
    If cell.Column = CONF_PRN And _
       cell.Row >= CONF_START And cell.Row <= CONF_END Then
        If Trim(CStr(Me.Cells(cell.Row, 1).Value)) <> "" Then
            Application.EnableEvents = False
            cell.Offset(0, -1).Select
            Application.EnableEvents = True
            Call ExportTotalHeadPRN(cell.Row)
        End If
    End If

End Sub
