' ============================================================
' VBA_TotalHead_Sheet.txt  -  Sheet module for "Total Head"
' v1.83
'
' Paste this entire file into the Sheet module for the
' "Total Head" worksheet in the VBA editor (Alt+F11,
' double-click "Total Head" in the project tree).
'
' CHANGE HANDLERS
'   D3  (Zone Name)   – auto-load from archive if name matches;
'                       clear zone notes on new name; update chart;
'                       clearing D3 auto-archives then wipes ONLY the
'                       pink cells: Driving Head (L3) and Zone Notes
'                       (J9).  Sensor config rows are NOT cleared.
'                       Start/End dates and System are also preserved.
'   D5:D6 (dates)     – rebuild data table + refresh chart
'   Col A (Point Ref) – record old note, populate Z/GPS/Dip/Dumpy
'                       from PI, sync offset from PD, rebuild data
'   Col B (Offset)    – sync offset back to PD row 5, rebuild
'   Col C (GPS)       – write GPS back to Point Index, record note
'   Col D (Z)         – write elevation back to Point Index
'   Col E (Dip)       – write Dip back to Point Index, record note
'   Col F (Dumpy)     – write Dumpy back to Point Index, record note
'
' SELECTION HANDLER
'   I5  (Save button) – export TotalHeadChart as PNG (SaveTotalHeadChartAsPNG)
'   L5  (Load drpdn)  – trigger handled via Worksheet_Change L5
' ============================================================
Option Explicit

' Snapshots taken in SelectionChange so that Worksheet_Change
' can report old values in Point Index notes.
Private mOrigZByRow(9 To 43)   As Variant  ' Z before Point Ref change
Private mOrigRefByRow(9 To 43) As String   ' Point Ref before it is cleared
Private mOrigZoneName          As String   ' Zone Name before D3 is changed

Private Const LOAD_DROPDOWN_PLACEHOLDER As String = """Drop Down to load saved selections"""

' ============================================================
' Worksheet_Activate
' ============================================================
Private Sub Worksheet_Activate()
    ' Safety net: if a previous error left EnableEvents=False,
    ' activating this sheet restores it so all handlers work again.
    Application.EnableEvents = True

    ' If the user has cells copied (marching ants), any VBA cell write
    ' will silently cancel Excel copy mode.  Skip the rebuild so that
    ' the user can switch to this sheet and paste without losing their
    ' copy selection.  The data table was already built on the last
    ' full activation and is still valid.
    If Application.CutCopyMode <> False Then Exit Sub

    ' Snapshot current Point Refs + Z values so that Worksheet_Change
    ' can record the old ref in PI notes even when the user clears a
    ' ref without first clicking on the cell (e.g. after a zone load).
    Dim i As Long
    For i = 9 To 43
        mOrigRefByRow(i) = Trim(CStr(Me.Cells(i, 1).Value))
        mOrigZByRow(i)   = Me.Cells(i, 4).Value
    Next i
    mOrigZoneName = Trim(CStr(Me.Cells(3, 4).Value))

    Call InitTotalHeadSheet
    Call PopulateAllElevationsTH
    Call RebuildDataTable
    Call CreateTotalHeadChart
    Call RefreshLoadDropdown
End Sub

' ============================================================
' Worksheet_Change
' ============================================================
Private Sub Worksheet_Change(ByVal Target As Range)
    On Error GoTo EH

    ' ── D3 Zone Name intercept (must come before multi-cell guard) ──
    ' Zone Name (D3) is a merged cell spanning D3:I4 — when the user
    ' clears it, Target.Cells.Count = 12 (> 1), so the multi-cell
    ' guard below would Exit Sub before the zone-name handler is
    ' reached.  Use Intersect to catch it first regardless of merge size.
    Dim r As Long, c As Long
    If Not Intersect(Target, Me.Cells(3, 4)) Is Nothing Then
        r = 3
        c = 4
    ' ── Multi-cell change (paste / fill-down / range-clear) ─────
    ' All multi-cell changes are ignored UNLESS they touch col A
    ' config rows 9-43.  In that case, re-sync all offsets and
    ' elevations then rebuild – same effect as a single-cell entry.
    ' When anyConfig is False the handler falls through to Exit Sub
    ' and the change is silently ignored.
    ElseIf Target.Cells.Count > 1 Then
        Dim pc As Range
        Dim anyConfig As Boolean
        For Each pc In Target.Cells
            If pc.Column = 1 And pc.Row >= 9 And pc.Row <= 43 Then
                anyConfig = True
                Exit For
            End If
        Next pc
        If anyConfig Then
            Application.EnableEvents = False
            Call RepopulateOffsetsFromPressureData
            Call PopulateAllElevationsTH
            Application.EnableEvents = True
            Call RebuildDataTable
            Call RefreshTotalHeadChart
        End If
        Exit Sub
    Else
        r = Target.Row
        c = Target.Column
    End If

    ' ── D3: Zone Name ───────────────────────────────────────
    If r = 3 And c = 4 Then
        Dim newZone As String
        newZone = Trim(CStr(Me.Cells(3, 4).Value))

        ' ── Zone Name cleared → auto-archive then wipe pink cells ─
        ' Only the "pink" zone-level cells are cleared:
        '   L3  – Driving Head
        '   J9  – Zone Notes
        ' Sensor config rows (Point Refs and their associated data)
        ' are intentionally preserved so they can be cleared
        ' individually by removing each Point Ref (which clears the
        ' "green" cells: Offset/GPS/Z/Dip/Dumpy/Notes for that row).
        If newZone = "" Then
            ' Archive current data first (so nothing is lost)
            If mOrigZoneName <> "" Then
                Call ArchiveTotalHeadData(mOrigZoneName)
            End If
            Application.EnableEvents = False
            Me.Cells(3, 12).Value = ""   ' Driving Head  (L3)  ← pink
            Me.Cells(9, 10).Value = ""   ' Zone Notes    (J9)  ← pink
            Application.EnableEvents = True
            mOrigZoneName = ""
            Call RebuildDataTable
            Call RefreshTotalHeadChart
            Exit Sub
        End If

        ' Auto-clear zone notes when zone name changes
        Application.EnableEvents = False
        Me.Cells(9, 10).Value = ""
        Application.EnableEvents = True

        ' Update chart title immediately
        On Error Resume Next
        Dim cht As ChartObject
        Set cht = Me.ChartObjects("TotalHeadChart")
        If Not cht Is Nothing And newZone <> "" Then
            cht.Chart.HasTitle = True
            cht.Chart.ChartTitle.Text = newZone
        End If
        On Error GoTo EH

        ' Check archive – load if this zone name is saved
        If newZone <> "" Then
            Dim wsArch As Worksheet
            Set wsArch = ThisWorkbook.Sheets("Total Head Archive")
            Dim lastArchRow As Long
            lastArchRow = wsArch.Cells(wsArch.Rows.Count, 3).End(xlUp).Row
            Dim rr As Long
            For rr = 2 To lastArchRow
                If Trim(CStr(wsArch.Cells(rr, 3).Value)) = newZone Then
                    Call LoadTotalHeadByName(newZone)
                    Exit Sub
                End If
            Next rr
        End If
        Exit Sub
    End If

    ' ── D5 / D6: Date range ─────────────────────────────────
    If c = 4 And (r = 5 Or r = 6) Then
        Call RebuildDataTable
        Call RefreshTotalHeadChart
        Exit Sub
    End If

    ' ── L5: Load dropdown ───────────────────────────────────
    If r = 5 And c = 12 Then
        Dim selectedZone As String
        selectedZone = Trim(CStr(Me.Cells(5, 12).Value))
        If selectedZone <> "" And selectedZone <> LOAD_DROPDOWN_PLACEHOLDER Then
            Call LoadTotalHeadByName(selectedZone)
        End If
        Exit Sub
    End If

    ' ── Col A: Point Ref (config rows 9-43) ─────────────────
    If c = 1 And r >= 9 And r <= 43 Then
        Dim ptRef As String
        ptRef = Trim(CStr(Me.Cells(r, 1).Value))

        If ptRef = "" Then
            ' Clearing a Point Ref:
            ' 1. Record note to PI first (while Z / Notes cells still have values).
            '    Wrapped in its own On Error Resume Next so a failure here NEVER
            '    prevents the cells from being cleared in step 2.
            If mOrigRefByRow(r) <> "" Then
                On Error Resume Next
                Call RecordSensorNote(Me, r, mOrigRefByRow(r), mOrigZByRow(r))
                On Error GoTo EH
            End If

            ' 2. Clear all associated cells (always runs, even if note recording failed).
            Application.EnableEvents = False
            Me.Cells(r, 2).Value = ""   ' Offset
            Me.Cells(r, 3).Value = ""   ' GPS
            Me.Cells(r, 4).Value = ""   ' Z
            Me.Cells(r, 5).Value = ""   ' Dip
            Me.Cells(r, 6).Value = ""   ' Dumpy
            Me.Cells(r, 8).Value = ""   ' Notes
            ' Clear data column header and data
            Dim dataCol As Long
            dataCol = r - 7
            Dim lastDataRow As Long
            lastDataRow = Me.Cells(Me.Rows.Count, 1).End(xlUp).Row
            If lastDataRow >= 46 Then
                Me.Range(Me.Cells(45, dataCol), Me.Cells(lastDataRow, dataCol)).ClearContents
            Else
                Me.Cells(45, dataCol).ClearContents
            End If
            Application.EnableEvents = True
            GoTo RebuildAndExit
        End If

        ' Snapshot current Z before it may be overwritten
        mOrigZByRow(r) = Me.Cells(r, 4).Value

        ' Record note ONLY when the Point Ref has actually changed
        ' (avoids spurious notes when the user re-enters the same ref).
        ' Wrapped in On Error Resume Next so a note-recording failure
        ' (e.g. Point Index write-protected) NEVER skips population.
        If ptRef <> mOrigRefByRow(r) Then
            On Error Resume Next
            Call RecordSensorNote(Me, r, ptRef, mOrigZByRow(r))
            On Error GoTo EH
        End If

        Application.EnableEvents = False

        ' Populate Z from Point Index
        Dim wsPI As Worksheet
        Set wsPI = ThisWorkbook.Sheets("Point Index")
        Call PopulateElevationTH(Me, wsPI, r)

        ' Populate GPS / Dip / Dumpy from Point Index
        Call PopulateGpsDipDumpyOneTH(Me, r)

        ' Sync offset from Pressure Data row 5
        Call SyncOffsetToTotalHead(ptRef, r)

        Application.EnableEvents = True
        GoTo RebuildAndExit
    End If

    ' ── Col B: Offset ────────────────────────────────────────
    If c = 2 And r >= 9 And r <= 43 Then
        Dim ptB As String
        ptB = Trim(CStr(Me.Cells(r, 1).Value))
        If ptB <> "" And IsNumeric(Me.Cells(r, 2).Value) Then
            Call SyncOffsetToPressureData(ptB, CDbl(Me.Cells(r, 2).Value))
        End If
        GoTo RebuildAndExit
    End If

    ' ── Col D: Z (m) ─────────────────────────────────────────
    If c = 4 And r >= 9 And r <= 43 Then
        Dim ptD As String
        ptD = Trim(CStr(Me.Cells(r, 1).Value))
        If ptD <> "" And IsNumeric(Me.Cells(r, 4).Value) Then
            Dim noteTimestamp As String
            noteTimestamp = Format(Now(), DATE_FMT) & " | Elevation Change"
            Call WriteElevationToPointIndex(ptD, CDbl(Me.Cells(r, 4).Value), noteTimestamp)
        End If
        GoTo RebuildAndExit  ' Z change needs rebuild (Final Z affects data table values)
    End If

    ' ── Col C: GPS ───────────────────────────────────────────
    If c = 3 And r >= 9 And r <= 43 Then
        Dim ptC As String
        ptC = Trim(CStr(Me.Cells(r, 1).Value))
        If ptC <> "" And IsNumeric(Me.Cells(r, 3).Value) Then
            Dim noteC As String
            noteC = Format(Now(), DATE_FMT) & " | GPS Change"
            Call WriteGpsToPointIndex(ptC, CDbl(Me.Cells(r, 3).Value), noteC)
        End If
        Exit Sub
    End If

    ' ── Col E: Dip ───────────────────────────────────────────
    If c = 5 And r >= 9 And r <= 43 Then
        Dim ptE As String
        ptE = Trim(CStr(Me.Cells(r, 1).Value))
        If ptE <> "" And IsNumeric(Me.Cells(r, 5).Value) Then
            Dim noteE As String
            noteE = Format(Now(), DATE_FMT) & " | Dip Change"
            Call WriteDipToPointIndex(ptE, CDbl(Me.Cells(r, 5).Value), noteE)
        End If
        Exit Sub
    End If

    ' ── Col F: Dumpy ─────────────────────────────────────────
    If c = 6 And r >= 9 And r <= 43 Then
        Dim ptF As String
        ptF = Trim(CStr(Me.Cells(r, 1).Value))
        If ptF <> "" And IsNumeric(Me.Cells(r, 6).Value) Then
            Dim noteF As String
            noteF = Format(Now(), DATE_FMT) & " | Dumpy Change"
            Call WriteDumpyToPointIndex(ptF, CDbl(Me.Cells(r, 6).Value), noteF)
        End If
        Exit Sub
    End If

    Exit Sub

RebuildAndExit:
    Call RebuildDataTable
    Call RefreshTotalHeadChart
    Exit Sub

EH:
    ' Any error while EnableEvents=False would leave it permanently
    ' False for the session – guarantee it is always restored.
    Application.EnableEvents = True
End Sub

' ============================================================
' Worksheet_SelectionChange
' 1. Detects a click on I5 (Save button, col 9, row 5).
' 2. Snapshots Point Ref + Z for config rows so Worksheet_Change
'    can reference the old value after the user edits the cell.
' ============================================================
Private Sub Worksheet_SelectionChange(ByVal Target As Range)
    On Error GoTo EH
    ' ── Snapshot col A config cells ─────────────────────────
    Dim sc As Range
    For Each sc In Target.Cells
        If sc.Column = 1 And sc.Row >= 9 And sc.Row <= 43 Then
            mOrigRefByRow(sc.Row) = Trim(CStr(sc.Value))
            mOrigZByRow(sc.Row)   = Me.Cells(sc.Row, 4).Value
        End If
        ' Snapshot Zone Name so that clearing D3 can archive with the old name
        If sc.Row = 3 And sc.Column = 4 Then
            mOrigZoneName = Trim(CStr(Me.Cells(3, 4).Value))
        End If
    Next sc

    ' ── Save button (I5) ────────────────────────────────────
    If Target.Row = 5 And Target.Column = 9 Then
        ' Save button: export chart as PNG
        Call SaveTotalHeadChartAsPNG
        ' Move selection away so a second click still triggers
        Me.Cells(3, 4).Select
    End If
    Exit Sub

EH:
    Application.EnableEvents = True
End Sub
